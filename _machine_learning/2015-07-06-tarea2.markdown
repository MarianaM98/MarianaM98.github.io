def perceptron(S,y,m,d):
    w=[0]*(d+1)
    for i in range(0,m-1):
        t=0
        for j in range(0,d-1):
            t=t+w[j+1]*S[i][j]
        t=t+w[0]
        if t!=y[i]:
            for j in range(0,d-1):
                S[i][j]=S[i][j]*y[i]
                w[j+1]=w[j+1]+S[i][j]
            w[0]=w[0]*y[i]
        else:
            return(w)
