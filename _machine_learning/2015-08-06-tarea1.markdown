def entrenamiento(k): 
    x=np.random.uniform(0,2,k)
    y=np.random.uniform(0,2,k)
    z=np.zeros(k)
    p=[]
    for i in range(0,k):
        if x[i]<(1+1/sqrt(2)) and x[i]>(1-1/sqrt(2)):
            if y[i]<(1+1/sqrt(2)) and y[i]>(1-1/sqrt(2)):
                z[i]=1
            else:
                z[i]=0
        else: 
            z[i]=0
    for i in range(0,k):
        p=p+[x[i],y[i],z[i]]
    return p
